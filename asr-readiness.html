How to read rollout status
Coverage bar = Block / Audit / Off. If Audit coverage is 50%+ with low hits, the rule is marked Ready candidate. If Block coverage is 90%+ with low audit impact, the rule is marked Finish rollout. If Off coverage is above 50%, the rule is marked Needs better rollout.

let Clients =
DeviceInfo
| where OSPlatform in ("Windows10", "Windows11")
| where OnboardingStatus == "Onboarded"
| summarize arg_max(Timestamp, *) by DeviceName
| project DeviceName, OSPlatform;
let AsrConfig =
Clients
| join kind=leftouter (
    DeviceTvmInfoGathering
    | summarize arg_max(Timestamp, *) by DeviceName
    | extend ASR = parse_json(AdditionalFields.AsrConfigurationStates)
    | project DeviceName, ASR
) on DeviceName
| extend ConfigKeys = bag_keys(ASR)
| mv-expand ConfigKeyDyn = ConfigKeys
| extend ConfigKey = tostring(ConfigKeyDyn)
| extend ConfigMode = tostring(ASR[ConfigKey])
| join kind=inner AsrRules on ConfigKey
| summarize
    TotalDevices = dcount(DeviceName),
    BlockConfiguredDevices = dcountif(DeviceName, ConfigMode == "Block"),
    AuditConfiguredDevices = dcountif(DeviceName, ConfigMode == "Audit"),
    OffDevices = dcountif(DeviceName, ConfigMode == "Off")
    by RuleId, RuleName, ConfigKey;
let AsrEvents =
DeviceEvents
| where DeviceName in (Clients | project DeviceName)
| where ActionType startswith "Asr"
| where ActionType contains "Audited" or ActionType contains "Blocked"
| extend AF = parse_json(AdditionalFields)
| extend RuleId = tolower(tostring(AF.RuleId))
| extend Mode = iff(ActionType contains "Blocked", "Block", "Audit")
| summarize
    TotalHits = count(),
    BlockedCount = countif(Mode == "Block"),
    AuditedCount = countif(Mode == "Audit"),
    AffectedDevices = dcount(DeviceName),
    AffectedDeviceList = make_set(DeviceName, 100),
    UniqueProcesses = dcount(InitiatingProcessFileName),
    SampleProcesses = make_set(InitiatingProcessFileName, 10),
    FirstSeen = min(Timestamp),
    LastSeen = max(Timestamp)
    by RuleId;
AsrRules
| join kind=leftouter AsrConfig on RuleId, RuleName, ConfigKey
| join kind=leftouter AsrEvents on RuleId
| extend
    TotalDevices = coalesce(TotalDevices, 0),
    BlockConfiguredDevices = coalesce(BlockConfiguredDevices, 0),
    AuditConfiguredDevices = coalesce(AuditConfiguredDevices, 0),
    OffDevices = coalesce(OffDevices, 0),
    BlockPct = round(100.0 * BlockConfiguredDevices / TotalDevices, 1),
    AuditPct = round(100.0 * AuditConfiguredDevices / TotalDevices, 1),
    OffPct = round(100.0 * OffDevices / TotalDevices, 1),
    TotalHits = coalesce(TotalHits, 0),
    BlockedCount = coalesce(BlockedCount, 0),
    AuditedCount = coalesce(AuditedCount, 0),
    AffectedDevices = coalesce(AffectedDevices, 0),
    UniqueProcesses = coalesce(UniqueProcesses, 0)
| extend RemainingToBlock = TotalDevices - BlockConfiguredDevices
| extend Status = case(
    AuditedCount > 50, "High attention",
    OffPct > 50, "Needs better rollout",
    BlockPct >= 99.95, "Enforced/Monitor",
    BlockPct >= 90 and AuditedCount <= 50, "Finish rollout",
    AuditPct >= 50 and AuditedCount <= 50, "Ready candidate",
    BlockPct > 0 and OffPct > 0, "Needs better rollout",
    AuditConfiguredDevices > 0 or AuditPct > 0, "Ready candidate",
    BlockConfiguredDevices > 0 and OffPct == 0 and AuditPct == 0, "Enforced/Monitor",
    "Review"
)
| extend Recommendation = case(
    OffPct > 50, "More than 50% of devices are Off for this rule — needs better rollout before readiness decision",
    BlockPct >= 99.95, "Block mode is fully deployed — monitor for false positives",
    BlockPct >= 90 and AuditedCount <= 50, strcat("Almost fully in Block mode — finish blocking the remaining ", tostring(RemainingToBlock), " Windows endpoint(s)"),
    AuditedCount > 50, "Audit enabled with significant hits — investigate before blocking",
    AuditConfiguredDevices > 0 and AuditedCount > 0, "Audit enabled with low hits — candidate for block after review",
    AuditConfiguredDevices > 0 and AuditedCount == 0, "Audit enabled with no hits — likely good block candidate",
    BlockConfiguredDevices > 0 and BlockedCount > 0, "Block enabled and enforcing",
    BlockConfiguredDevices > 0 and BlockedCount == 0, "Block enabled, no matching activity seen",
    "Review configuration"
)
| project RuleName, ConfigKey, TotalDevices, BlockConfiguredDevices, AuditConfiguredDevices, OffDevices, BlockPct, AuditPct, OffPct, Status, TotalHits, BlockedCount, AuditedCount, AffectedDevices, UniqueProcesses, Recommendation, SampleProcesses, AffectedDeviceList, FirstSeen, LastSeen
| order by Status asc, AuditedCount desc, AffectedDevices desc
